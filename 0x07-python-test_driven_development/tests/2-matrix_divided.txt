# Test Case 1: Dividing a matrix by a non-zero divisor
Input:
matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]
div = 2

Expected Output:
Resulting matrix:
[
    [0.5, 1.0, 1.5],
    [2.0, 2.5, 3.0],
    [3.5, 4.0, 4.5]
]

# Test Case 2: Dividing a matrix by zero
Input:
matrix = [
    [1, 2],
    [3, 4]
]
div = 0

Expected Output:
Expected Exception: ZeroDivisionError("division by zero")

# Test Case 3: Dividing a matrix with non-numeric elements
Input:
matrix = [
    [1, 2, 3],
    [4, 'a', 6],
    [7, 8, 9]
]
div = 2

Expected Output:
Expected Exception: TypeError("matrix must be a matrix (list of lists) of integers/floats")

# Test Case 4: Dividing a matrix with rows of different sizes
Input:
matrix = [
    [1, 2, 3],
    [4, 5],
    [7, 8, 9]
]
div = 2

Expected Output:
Expected Exception: TypeError("Each row of the matrix must have the same size")

# Test Case 5: Dividing an empty matrix
Input:
matrix = []
div = 2

Expected Output:
Resulting matrix: []

# Test Case 6: Dividing a matrix by a negative divisor
Input:
matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]
div = -2

Expected Output:
Resulting matrix:
[
    [-0.5, -1.0, -1.5],
    [-2.0, -2.5, -3.0],
    [-3.5, -4.0, -4.5]
]
